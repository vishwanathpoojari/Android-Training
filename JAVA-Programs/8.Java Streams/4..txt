import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class Test{
    public static void main(String[] args) {
        List<String> stringListWithDuplicates = Arrays.asList("apple", "orange", "banana", "apple", "grape", "banana");
        List<String> uniqueStringList = removeDuplicates(stringListWithDuplicates);
        System.out.println("Original List with Duplicates: " + stringListWithDuplicates);
        System.out.println("List without Duplicates: " + uniqueStringList);
    }
    private static <T> List<T> removeDuplicates(List<T> listWithDuplicates) {
        return listWithDuplicates.stream()
                .distinct()
                .collect(Collectors.toList());
    }
}
