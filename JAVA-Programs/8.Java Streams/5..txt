import java.util.Arrays;
import java.util.List;

public class Test{
    public static void main(String[] args) {
        List<String> stringList = Arrays.asList("apple", "banana", "orange", "avocado", "grape", "kiwi");
        char specificLetter = 'a';
        long count = countStringsStartingWithLetter(stringList, specificLetter);
        System.out.println("Original List: " + stringList);
        System.out.println("Number of strings starting with '" + specificLetter + "': " + count);
    }
    private static long countStringsStartingWithLetter(List<String> strings, char letter) {
        return strings.stream()
                .filter(s -> s.toLowerCase().startsWith(String.valueOf(letter).toLowerCase()))
                .count();
    }
}
