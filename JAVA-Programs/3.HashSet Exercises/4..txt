import java.util.HashSet;

public class Example{
    public static void main(String[] args) {

        HashSet<String> hashSet = new HashSet<>();

        hashSet.add("Element 1");
        hashSet.add("Element 2");
        hashSet.add("Element 3");

        System.out.println("HashSet before emptying: " + hashSet);
        example(hashSet);
        System.out.println("HashSet after emptying: " + hashSet);
    }
    private static void example(HashSet<String> hashSet) {
        // Using the clear() method to remove all elements from the hash set
        hashSet.clear();
    }
}
