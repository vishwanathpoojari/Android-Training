import java.util.HashSet;

public class Example {
    public static void main(String[] args) {
        HashSet<String> hashSet = new HashSet<>();

        hashSet.add("Element 1");
        hashSet.add("Element 2");
        hashSet.add("Element 3");

        System.out.println("HashSet before appending: " + hashSet);
        String elementToAppend = "New Element";
        example(hashSet, elementToAppend);
        System.out.println("HashSet after appending: " + hashSet);
    }


    private static void example(HashSet<String> hashSet, String element) {
        hashSet.add(element);
    }
}
