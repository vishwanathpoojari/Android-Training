import java.util.HashSet;

public class Test {
    public static void main(String[] args) {

        HashSet<String> hashSet = new HashSet<>();
        boolean isEmpty = isHashSetEmpty(hashSet);
        System.out.println("Is the HashSet empty? " + isEmpty);

        hashSet.add("Element 1");
        hashSet.add("Element 2");
        hashSet.add("Element 3");

        isEmpty = isHashSetEmpty(hashSet);
        System.out.println("Is the HashSet empty now? " + isEmpty);
    }


    private static boolean isHashSetEmpty(HashSet<String> hashSet) {
        return hashSet.isEmpty();
    }
}
