import java.util.HashSet;

public class Example{
    public static void main(String[] args) {

        HashSet<String> hashSet = new HashSet<>();

        hashSet.add("Element 1");
        hashSet.add("Element 2");
        hashSet.add("Element 3");
        int numberOfElements = getHashSetSize(hashSet);
        System.out.println("Number of elements in the HashSet: " + numberOfElements);
    }

    private static int getHashSetSize(HashSet<String> hashSet) {
        return hashSet.size();
    }
}
