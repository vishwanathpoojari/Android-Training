import java.util.Scanner;

public class LastNVowelsExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String inputString = scanner.nextLine();
        System.out.print("Enter the value of n: ");
        int n = scanner.nextInt();
        String lastNVowels = getLastNVowels(inputString, n);
        System.out.println("Original String: " + inputString);
        System.out.println("Last " + n + " Vowels: " + lastNVowels);
    }

    private static String getLastNVowels(String input, int n) {
        StringBuilder lastVowels = new StringBuilder();
        int vowelCount = 0;
        for (int i = input.length() - 1; i >= 0 && vowelCount < n; i--) {
            char currentChar = Character.toLowerCase(input.charAt(i));

            if (isVowel(currentChar)) {
                lastVowels.insert(0, input.charAt(i));  // Insert at the beginning to maintain order
                vowelCount++;
            }
        }

        return lastVowels.toString();
    }

    private static boolean isVowel(char c) {
        return c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u';
    }
}
