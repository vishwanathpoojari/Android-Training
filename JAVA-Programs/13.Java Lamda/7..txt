import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

public class LambdaRemoveDuplicatesExample {
    public static void main(String[] args) {
        List<Integer> integerList = Arrays.asList(1, 2, 3, 2, 4, 5, 6, 1, 7, 8, 5);
        List<Integer> distinctList = integerList.stream().distinct().collect(Collectors.toList());
        System.out.println("Original List: " + integerList);
        System.out.println("List without duplicates: " + distinctList);
    }
}
