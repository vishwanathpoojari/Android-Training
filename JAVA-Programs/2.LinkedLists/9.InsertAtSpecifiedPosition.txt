import java.util.LinkedList;

public class InsertAtPositionLinkedList {
    public static void main(String[] args) {
        LinkedList<String> linkedList = new LinkedList<>();
        linkedList.add("Element 1");
        linkedList.add("Element 2");
        linkedList.add("Element 3");
        System.out.println("Linked List before insertion: " + linkedList);
        String element1 = "New Element 1";
        String element2 = "New Element 2";
        int position1 = 1;
        int position2 = 3;
        insertAtPosition(linkedList, position1, element1);
        insertAtPosition(linkedList, position2, element2);
        System.out.println("Linked List after insertion: " + linkedList);
    }
    private static void insertAtPosition(LinkedList<String> linkedList, int position, String element) {
        if (position >= 0 && position <= linkedList.size()) {
            linkedList.add(position, element);
        } else {
            System.out.println("Invalid position. Element not inserted.");
        }
    }
}
