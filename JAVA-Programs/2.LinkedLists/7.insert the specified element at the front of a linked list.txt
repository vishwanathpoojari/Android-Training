import java.util.LinkedList;

public class InsertAtFrontLinkedList {
    public static void main(String[] args) {
        LinkedList<String> linkedList = new LinkedList<>();
        linkedList.add("Element 1");
        linkedList.add("Element 2");
        linkedList.add("Element 3");
        System.out.println("Linked List before insertion: " + linkedList);
        String elementToInsert = "New Element";
        insertAtFront(linkedList, elementToInsert);
        System.out.println("Linked List after insertion: " + linkedList);
    }
    private static void insertAtFront(LinkedList<String> linkedList, String element) {
        linkedList.addFirst(element);
    }
}
