import java.util.TreeSet;

public class Test {
    public static void main(String[] args) {

        TreeSet<String> treeSet = new TreeSet<>();
        treeSet.add("Red");
        treeSet.add("Green");
        treeSet.add("Blue");
        treeSet.add("Yellow");
        treeSet.add("Purple");

        System.out.println("Original TreeSet: " + treeSet);
        String firstElement = getFirstElement(treeSet);
        String lastElement = getLastElement(treeSet);
        System.out.println("First Element: " + firstElement);
        System.out.println("Last Element: " + lastElement);
    }

    private static String getFirstElement(TreeSet<String> originalSet) {
        return originalSet.first();
    }

    private static String getLastElement(TreeSet<String> originalSet) {
        return originalSet.last();
    }
}
