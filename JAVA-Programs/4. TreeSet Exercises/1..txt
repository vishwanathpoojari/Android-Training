import java.util.TreeSet;

public class Example {
    public static void main(String[] args) {

        TreeSet<String> treeSet = new TreeSet<>();
        treeSet.add("Red");
        treeSet.add("Green");
        treeSet.add("Blue");
        treeSet.add("Yellow");
        treeSet.add("Purple");
        System.out.println("TreeSet of Colors: " + treeSet);
    }
}
