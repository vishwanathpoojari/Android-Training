import java.util.TreeSet;
import java.util.Iterator;

public class Example {
    public static void main(String[] args) {

        TreeSet<String> treeSet = new TreeSet<>();
        treeSet.add("Red");
        treeSet.add("Green");
        treeSet.add("Blue");
        treeSet.add("Yellow");
        treeSet.add("Purple");
        System.out.println("Original TreeSet: " + treeSet);
        TreeSet<String> reverseOrderSet = createReverseOrderView(treeSet);
        System.out.println("Reverse Order View: " + reverseOrderSet);
    }

    private static TreeSet<String> createReverseOrderView(TreeSet<String> originalSet) {
        // Using the descendingSet() method to create a reverse order view
        return (TreeSet<String>) originalSet.descendingSet();
    }
}
