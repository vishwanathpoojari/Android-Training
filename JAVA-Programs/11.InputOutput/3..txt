import java.nio.file.*;

public class FileExistenceExample {
    public static void main(String[] args) {
        String path = "/path/to/your/file_or_directory";
        boolean exists = checkExistence(path);
        if (exists) {
            System.out.println("The file or directory exists.");
        } else {
            System.out.println("The file or directory does not exist.");
        }
    }

    private static boolean checkExistence(String path) {
        Path filePath = Paths.get(path);
        return Files.exists(filePath);
    }
}
