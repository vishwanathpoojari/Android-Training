import java.util.HashMap;
import java.util.Map;

public class Test{
    public static void main(String[] args) {
        Map<String, Integer> myMap = new HashMap<>();
        myMap.put("viswa", 25);
        myMap.put("Kethan", 30);
        myMap.put("Bob", 22);
        System.out.println("Initial size of the map: " + myMap.size());
        int numberOfMappings = countMappings(myMap);
        System.out.println("Number of key-value mappings: " + numberOfMappings);
    }
    private static <K, V> int countMappings(Map<K, V> map) {
        return map.size();
    }
}
